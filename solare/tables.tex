\setupxtable[
  frame=off,
  split=yes,
  header=repeat,
  option={fit, width, stretch},
]

\setupxtable[head][
  topframe=on,
  bottomframe=on,
  foregroundstyle=bold,
  background=color,
  backgroundcolor=TextColourPrimaryLt,
]

\setupxtable[body][]

\setupxtable[foot][
  bottomframe=on,
]
