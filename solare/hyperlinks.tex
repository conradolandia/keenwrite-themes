% Avoid green hyperlinks.
\setupinteraction[
  state=start,
  color=BookColourHyperlink,
  contrastcolor=BookColourHyperlink,
  style=,
]

\setupurl[style=\txx, space=yes]

% Create hyperlink references within the document.
\define[2]\href{%
  \goto{\tf\color[BookColourHyperlink]{#1}}[url(#2)]%
}

\define[1]\email{%
  \goto{\tf\color[BookColourHyperlink]{#1}}[url(mailto:#1)]%
}

\define[1]\phone{%
  \goto{\tf\color[BookColourHyperlink]{#1}}[url(tel:#1)]%
}
