% Set up the body and heading fonts.
\starttypescript [serif] [TextFontBody]
  \definefontsynonym[Serif]
    [name:calibriregular]
    [features=default]
  \definefontsynonym[SerifBold]
    [name:calibribold]
    [features=default]
  \definefontsynonym[SerifItalic]
    [name:calibriitalic]
    [features=default]
  \definefontsynonym[SerifBoldItalic]
    [name:calibribolditalic]
    [features=default]
\stoptypescript

\starttypescript [sans] [TextFontHeading]
  \definefontsynonym[Sans]
    [name:bryantproregular]
    [features=default]
  \definefontsynonym[SansBold]
    [name:bryantprobold]
    [features=default]
  \definefontsynonym[SansLight]
    [name:bryantprolight]
    [features=default]
  \definefontsynonym[SansRegular]
    [name:bryantproregular]
    [features=default]
\stoptypescript

\definetypeface[TextFont] [rm] [serif] [TextFontBody]    [default]
\definetypeface[TextFont] [ss] [sans]  [TextFontHeading] [default]
\definetypeface[TextFont] [mm] [math]  [schola]          [default] [rscale=auto]
\definetypeface[TextFont] [tt] [mono]  [cursor]          [default] [rscale=0.85]

\usetypescript[TextFont]
\setupbodyfont[TextFont]

% Equate \it with \em
\setupbodyfontenvironment[default][em=italic]

% Font sizes for chapter heading and chapter number.
\definefont[TextFontChapter][SansRegular sa 2.5]
\definefont[TextFontSection][SansRegular sa 1.75]
\definefont[TextFontSubsection][SansRegular sa 1.25]
