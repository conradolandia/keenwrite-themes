% %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%% 
%
% Configure section headings, including chapters.
%
% %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%% 

\define[2]\TextChapter{\midaligned{Chapter #1 --- #2}}
\define[1]\TextByLine{\noindentation by #1}

\setuphead[
  align={nothyphenated, notcollapsed, verytolerant, broad},
]

\definenumber[TextSubsection][
  prefix=no,
]

\setuphead[chapter][
  style=,
  command=\TextChapter,
  before={\blank[force 6*big]},
  after={\blank[2*medium]},
  align=middle,
]

\setuphead[section][
  number=no,
  style=\Words,
  before={\blank[2*big]},
  after={\blank[2*medium]},
]

\setuphead[subsection][
  number=no,
  style=\Words,
  indent=no,
  before={
    % Add a scene separator before beginning the second subsection.
    \incrementnumber[TextSubsection]
    \doif{\rawcountervalue[TextSubsection]}{2}{\thinrule}
  },
  deeptextcommand=\TextByLine,
]
