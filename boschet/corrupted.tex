% Configure corrupted page style.
%
% %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%% 

\definestartstop[corrupted][
  before={%
    \startTEXpage[
      offset=\backspace,
      width=\paperwidth,
      height=\paperheight
    ]%
    \incrementnumber[userpage]
    \blank[big]%
  },
  after={\stopTEXpage},
]
